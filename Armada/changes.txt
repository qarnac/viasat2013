function controlMusic()
{
	if (music.isOn) 
	{ 
		music.isOn = 0; 
		muteMusic.innerHTML = "Unmute music"; 
		if (!shootSound.isOn) {	muteAll.innerHTML = "Unmute all"; }

	}
	else 
	{
		music.isOn = 1; 
		muteMusic.innerHTML = "Mute music"; 
		muteAll.innerHTML = "Mute all";
	}

	music.volume = .1 * music.isOn;	
}
